.concat()

        Array 
        Array1.concat(Array2)

        .concat() will return a new array consisting of the combined array contents.
        .concat() can take multiple Arrays 

        String
        str1.concat(", " ,str2) || str1.concat(str2) || str1.concat(str2, str3, str4)

        .concat() will return a new String consisting of the combined array contents.
        + or += is recommended over .concat() when used with strings.

.every()

        Array
        const Array1 = [35, 68, 90]
        let value = Array1.every((element, index, array)=>{
                    element > 20
                    })
        console.log(value) // true
        
        .every() will return true if all elements in the array pass the test.
        Else .every() will return false.

.some()

        Array
        const Array1 = [35, 68, 90]
        let value = Array1.some((element, index, array)=>{
                    element > 20
                    })
        console.log(value) // true
        
        .some() will return true if at least one element passes the test.
        Else .some() will return false.

.filter()

        Array
        const Array1 = [35, 68, 90]
        let value = Array1.filter((element, index, array)=>{
                    element > 56
                    })

        console.log(value) // [68, 90]

        .filter() does not modify current array and does return a new array with the values that pass the tests.

.forEach()

        Array
        const Array1 = [35, 68, 90]
        let hold = []
        Array1.forEach((element, index, array) =>{
                    Hold.push(element + 2)
                    })

        console.log(hold) // [37, 70, 92]

        .forEach() calls a function for each item in an array.

.indexOf()

        Array || String
        
        let returnedIndex = Array.indexOf(element, startFromIndex)

        const Array1 = [35, 68, 90]
        let index = Array1.indexOf(68)
        console.log(index) // 1

        .indexOf() returns the first index location for the value entered. If there is no match it will return -1.
        .indexOf() is like lastIndexOf but gets first index instead of last

.join()

        Array || String
        
        Array.join(separator)

        const Array1 = [35, 68, 90]

        let returnVallue = Array1.join()
        let otherVal = Array1.join(" ")

        console.log(returnVallue) // "35,68,90"
        console.log(otherVal) // "35 68 90"

        .join() turns an array into a string joined by the chosen separator.

.lastIndexOf()

        Array || String
        
        let returnedIndex = Array.lastIndexOf(element, startFromIndex)

        const Array1 = [35, 68, 90, 68]
        let index = Array1.lastIndexOf(68)
        console.log(index) // 3

        .lastIndexOf() returns the last index location for the value entered. If there is no match it will return -1. 
        .lastIndexOf() is like indexOf but gets last index instead of first

.map()

        Array || String
        
        let returnedArray = Array.map(element, index, array)

        const Array1 = [35, 68, 90, 68]
        let newArray = Array1.map(element => element + 2)
        console.log(newArray) // [37, 70, 92, 70]

        .map() returns a new array having ran a function against every element.
        .map() is like forEach but returns a new array after running a function against every element where .forEach() simpy runs a function against every element.

.pop()

        Array || String
        
        Array.pop()

        const Array1 = [35, 68, 90, 70]
        let popped = Array1.pop()
        console.log(Array1) // [35, 68, 90]
        console.log(popped) // 70

        .pop() removes the last index of an array.
        .pop() mutates the original array.

.push()

        Array || String
        let Array = []
        Array.push(value)

        const Array1 = [35, 68, 90, 70]
        Array1.push(100)
        console.log(Array1) // [35, 68, 90, 70, 100]
        

        .push() adds the provided value as the last index of an array.
        .push() mutates the original array.

.shift()

        Array || String
        Array.shift()

        const Array1 = [1, 2, 3, 4]
        Array1.shift() 
        console.log(Array1) // [2, 3, 4]
        
        .shift() is like .pop() but removes the first index in an array where .pop() removes the last index.
        .shift() mutates the original array.

.slice()

        Array || String
        Array.slice(start, end)

        const Array1 = [1, 2, 3, 4]
        let returnedArr = Array1.slice(1, 3) 
        console.log(returnedArr) // [2, 3]
        
        .slice() returns an array starting at the profided start param up to (not including) the last provided index or if no ending index
        is provided, it will return the rest of said array.
        .slice() does not mutate the original array.

.unshift()

        Array || String
        Array.unshift(value)

        const Array1 = [1, 3000, 90, 4]
         Array1.unshift("Texas") 
        console.log(Array1) // ["Texas", 1, 3000, 90, 4]
        
        .unshift() is like .push() but adds to the front of an array where .push() adds to the end of an array.
        .unshift() mutates the original array.
